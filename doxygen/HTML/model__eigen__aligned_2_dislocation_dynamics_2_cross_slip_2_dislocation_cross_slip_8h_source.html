60  CrossSlipContainerType findCrossSlipSegments(const DislocatedMaterial<dim,Isotropic>& material,
61  const int& crossSlipModel) const
62  {
63 
64  const double sinCrossSlipRad(std::sin(crossSlipDeg*M_PI/180.0));
65  CrossSlipContainerType crossSlipDeq;
66 
67  for(const auto& link : DN.links())
68  {
69  if( !link.second->isBoundarySegment()
70  && !link.second->source->isBoundaryNode()
71  && !link.second->sink->isBoundaryNode()
72  && !link.second->isGrainBoundarySegment()
73  && !link.second->source->isGrainBoundaryNode()
74  && !link.second->sink->isGrainBoundaryNode()
75  && !link.second->hasZeroBurgers()
76  && link.second->isGlissile()
77  && link.second->chord().normalized().cross(link.second->burgers().normalized()).norm()<=sinCrossSlipRad
78  && link.second->chord().norm()>2.0*DN.networkRemesher.Lmin
79  )
80  {
81 // const auto& grain(**link.second->grains().begin());
82 
83 
84  if(material.crystalStructure=="BCC")
85  {
86  CrossSlipModels<BCClattice<dim>>::addToCrossSlip(*link.second,crossSlipDeq,crossSlipModel);
87  }
88  else if(material.crystalStructure=="FCC")
89  {
90  CrossSlipModels<FCClattice<dim>>::addToCrossSlip(*link.second,crossSlipDeq,crossSlipModel);
91  }
92  else
93  {
94  std::cout<<"Unknown cross-slip model for crystal structure '"<<material.crystalStructure<<"'. Exiting."<<std::endl;
95  exit(EXIT_FAILURE);
96  }

98  }
99  }

101  return crossSlipDeq;
