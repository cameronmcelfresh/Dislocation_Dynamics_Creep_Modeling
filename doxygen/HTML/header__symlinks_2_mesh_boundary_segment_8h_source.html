93  void emplace_unique(const VectorDim& P0,const VectorDim& P1,const PlanarMeshFace<dim>* const face)
94  {
95  if((P0-P1).norm()>FLT_EPSILON)
96  {// ignore degenerate segments
97  int merged(0);
98  for(auto& seg : *this)
99  {
100  if( ((seg->P0-P0).norm()<FLT_EPSILON && (seg->P1-P1).norm()<FLT_EPSILON)
101  || ((seg->P0-P1).norm()<FLT_EPSILON && (seg->P1-P0).norm()<FLT_EPSILON)
102  )
103  {// coincident segments on different faces, need to merge faces
104  FaceContainerType mergedFaces(seg->faces);
105  mergedFaces.insert(face);
106  seg.reset(new MeshBoundarySegment<dim>(P0,P1,mergedFaces));
107  merged++;
108  }
109  }
110  if (merged)
111  {// P0->P1 was found to be not unique. Swap vectors
112  assert(merged==1 && "MORE THAN ONE SEGMENTS FOUND ON LINE P0 P1");
113  }
114  else
115  {// P0->P1 was found to be unique, add it
116  this->emplace_back(new MeshBoundarySegment<dim>(P0,P1,face));
117  }
118  }
119  }

150  const double segLength((P1-P0).norm());

152  // Compute intersection between boundary segment and line of intersection of the two planes
153  LineLineIntersection<dim> lli(P0,D0,ppi.P,ppi.d);
154  if(lli.type==LineLineIntersection<dim>::INCIDENT)
155  {
156  const double u0((lli.x0-P0).dot(D0));
157  if(u0>0.0-FLT_EPSILON && u0<segLength+FLT_EPSILON)

160  }
161  }

164  roots.insert(P0);
165  roots.insert(P1);
