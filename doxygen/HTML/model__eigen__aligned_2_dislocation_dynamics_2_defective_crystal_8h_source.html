264  simulationParameters.totalTime+=simulationParameters.dt;

273  {
274  PlanarDislocationSuperLoop<typename DislocationNetworkType::LoopType> superLoop(*loop.second);

278  // move

280 
281  // menage discrete topological events
282  DN->singleGlideStepDiscreteEvents(simulationParameters.runID);
