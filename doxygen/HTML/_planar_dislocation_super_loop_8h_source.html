131  static MatrixDim getL2G(const VectorDim& x,
132  const VectorDim& z)
133  {
134  const double xNorm(x.norm());
135  const double zNorm(z.norm());
136  assert(xNorm>FLT_EPSILON);
137  assert(zNorm>FLT_EPSILON);
138  assert(fabs(x.dot(z)<FLT_EPSILON*xNorm*zNorm));
139  MatrixDim temp(Eigen::Matrix3d::Identity());
140  temp.col(2)=z/zNorm;
141  temp.col(0)=x/xNorm;
142  temp.col(1)=temp.col(2).cross(temp.col(0));
143  return temp;
144  }
