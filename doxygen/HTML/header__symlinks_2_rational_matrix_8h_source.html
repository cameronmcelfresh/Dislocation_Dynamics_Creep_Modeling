48  {
49 
50  // Find the BestRationalApproximation of each entry
51  MatrixDimI nums(MatrixDimI::Zero());
52  MatrixDimI dens(MatrixDimI::Ones());
53 
54  IntValueType sigma=1;
55  for(int i=0;i<dim;++i)
56  {
57  for(int j=0;j<dim;++j)
58  {
59  BestRationalApproximation bra(R(i,j),maxDen);
60  nums(i,j)=bra.num;
61  dens(i,j)=bra.den;
62  sigma=lcm(sigma,bra.den);
63  }
64  }
65 
66  MatrixInt im(MatrixInt::Zero());
67  for(int i=0;i<dim;++i)
68  {
69  for(int j=0;j<dim;++j)
70  {
71  im(i,j)=nums(i,j)*sigma/dens(i,j);
72  }
73  }
74 
75  const double error=(im.template cast<double>()/sigma-R).norm()/(dim*dim);
76  if(error>100.0*DBL_EPSILON)

83  assert(false && "Rational Matrix failed, check maxDen.");
