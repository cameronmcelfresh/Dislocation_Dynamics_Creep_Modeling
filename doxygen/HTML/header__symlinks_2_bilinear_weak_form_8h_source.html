99  std::vector<Eigen::Triplet<double> > globalTriplets() const
100  {
101 
102  model::cout<<"Assembling BilinearWeakForm on domain..."<<std::flush;
103 
104  std::vector<Eigen::Triplet<double> > glbtrip;
105  glbtrip.reserve(dofPerElement*dofPerElement*TrialBase<TrialFunctionType>::elementSize());
106 
107  const auto t0= std::chrono::system_clock::now();
108  for (size_t k=0;k<domain.size();++k)
109  {
110  const ElementType& ele(domain.element(k));
111  ElementMatrixType ke(ElementMatrixType::Zero());
112  QuadratureType::integrate(this,ke,&BilinearWeakFormType::elementAssemblyKernel<AbscissaType>,ele);
113 
114  for (int i=0;i<dofPerElement;++i) //Assemble ke into global array of triplets
115  {
116  for (int j=0;j<dofPerElement;++j)
117  {
118  if(ke(i,j)!=0.0)
119  {
120  const size_t nodeID_I(i/dofPerNode);
121  const size_t nodeDof_I(i%dofPerNode);
122  const size_t gI= ele.node(nodeID_I).gID*dofPerNode+nodeDof_I;
123 
124  const size_t nodeID_J(j/dofPerNode);
125  const size_t nodeDof_J(j%dofPerNode);
126  const size_t gJ=ele.node(nodeID_J).gID*dofPerNode+nodeDof_J;
127 
128  glbtrip.emplace_back(gI,gJ,ke(i,j));
129  }
130  }
131  }
132  }
133 
134  model::cout<<" done.["<<(std::chrono::duration<double>(std::chrono::system_clock::now()-t0)).count()<<" sec]"<<std::endl;
135  return glbtrip;
