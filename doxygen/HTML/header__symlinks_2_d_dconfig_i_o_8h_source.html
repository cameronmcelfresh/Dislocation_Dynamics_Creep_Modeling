149  /**********************************************************************/
150  std::map<std::pair<size_t,size_t>,DislocationSegmentIO<dim>> segments() const
151  {
152 
153  std::map<std::pair<size_t,size_t>,DislocationSegmentIO<dim>> temp;
154 
155  for(const auto& link : links())
156  {
157 
158 
159  const auto loopIter=loopMap().find(link.loopID);
160  assert(loopIter!=loopMap().end());
161 
162  const size_t sourceID(std::min(link.sourceID,link.sinkID));
163  const size_t sinkID(std::max(link.sourceID,link.sinkID));
164  const auto key=std::make_pair(sourceID,sinkID);
165 
166  const auto insertPair=temp.insert(std::make_pair(key,DislocationSegmentIO<dim>(sourceID,sinkID)));
167  const auto& iter=insertPair.first;
168  const bool& success=insertPair.second;
169 
170  if(success)
171  {
172  iter->second.n=loopIter->second->N;
173  }
174  else
175  {
176  if(iter->second.n.cross(loopIter->second->N).norm()>FLT_EPSILON)
177  {
178  iter->second.n.setZero();
179  }
180  }
181 
182  if(link.sourceID<link.sinkID)
183  {
184  iter->second.b+=loopIter->second->B;
185  }
186  else
187  {
188  iter->second.b-=loopIter->second->B;
189  }
190 
191  if(iter->second.meshLocation==-1)
192  {
193  iter->second.meshLocation=link.meshLocation;
194  }
195  else
196  {
197  assert(iter->second.meshLocation==link.meshLocation);
198  }
199 
200  }
201 
202  return temp;
203  }
