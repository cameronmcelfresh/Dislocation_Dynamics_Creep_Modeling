117  /* */ const int& boundaryFace) const
118  {

130  Eigen::Matrix<double,6,6> get_C(const double& mu, const double& nu) const
131  {
132 // const double mu=1.0; // dimensionless
133 // const double nu=Material<dim,Isotropic>::nu;
134  const double lam=2.0*mu*nu/(1.0-2.0*nu);
135  const double C11(lam+2.0*mu);
136  const double C12(lam);
137  const double C44(mu); // C multiplies engineering strain
138 
139  Eigen::Matrix<double,6,6> temp;

142  /***/ C12, C12, C11, 0.0, 0.0, 0.0,
143  /***/ 0.0, 0.0, 0.0, C44, 0.0, 0.0,

156  Eigen::VectorXd solve(const Eigen::VectorXd& b,const Eigen::VectorXd& y)
157  {
161  model::cout<<"Setting up null-space solver..."<<std::flush;
162  const auto t0= std::chrono::system_clock::now();
163 
164  const size_t cSize(displacement().dirichletConditions().size());
165  const size_t tSize = gSize-cSize;
166 
167  std::vector<Eigen::Triplet<double> > tTriplets;
168  tTriplets.reserve(tSize);
169 
170  Eigen::VectorXd g(Eigen::VectorXd::Zero(gSize));
171  Eigen::VectorXd guess(Eigen::VectorXd::Zero(tSize));
172 
173  size_t startRow=0;
174  size_t col=0;
175 
176  for (const auto& cIter : displacement().dirichletConditions())
177  {
178 // for (typename DirichletConditionContainerType::const_iterator cIter =displacement().dirichletConditions().begin();
179 // /* */ cIter!=displacement().dirichletConditions().end();
180 // /* */ cIter++)
181 // {
182 // const size_t& endRow = cIter->first;
183 // g(endRow)= cIter->second;
184  const size_t& endRow = cIter.first;
185  g(endRow)= cIter.second;
186  for (size_t row=startRow;row!=endRow;++row)
187  {
188  tTriplets.emplace_back(row,col,1.0);
189  guess(col)=y(row);
190  col++;
191  }
192  startRow=endRow+1;
193  }
194  for (size_t row=startRow;row!=gSize;++row)
195  {
196  tTriplets.emplace_back(row,col,1.0);
197  guess(col)=y(row);
198  col++;
199  }
200 
201  SparseMatrixType T1(gSize,tSize);
202  T1.setFromTriplets(tTriplets.begin(),tTriplets.end());
203  model::cout<<" ["<<(std::chrono::duration<double>(std::chrono::system_clock::now()-t0)).count()<<" sec]"<<std::endl;
204 
205  // Check if null-space has changed
206  const auto t4= std::chrono::system_clock::now();
207  model::cout<<"Checking if null-space has changed... "<<std::flush;
208  bool sameT=false;
209  if(T.cols()==T1.cols() && T.rows()==T1.rows())
210  {
211  const double normT2=(T1-T).squaredNorm();
212  if(normT2==0.0)
213  {
214  sameT=true;
215  }
216 
217  }
218  model::cout<<!sameT<<std::flush;
219  model::cout<<" ["<<(std::chrono::duration<double>(std::chrono::system_clock::now()-t4)).count()<<" sec]"<<std::endl;
220 
221  // Update A1 if null-space has changed
222  if(!sameT)
223  { // need to re-factorized the LLT decomposition
224  const auto t1= std::chrono::system_clock::now();
225  if(use_directSolver)
226  {
227 #ifdef _MODEL_PARDISO_SOLVER_
228  model::cout<<"PardisoLLT: factorizing..."<<std::flush;
229 #else
230  model::cout<<"SimplicialLLT: factorizing..."<<std::flush;
231 #endif
232  }
233  else
234  {
235  model::cout<<"ConjugateGradient: factorizing..."<<std::flush;
236  }
237 
238  T=T1; // store new T
239  A1=T.transpose()*A*T; // store new A1
240  if(use_directSolver)
241  {
242  directSolver.compute(A1);
243  assert(directSolver.info()==Eigen::Success && "SOLVER FAILED");
244  }
245  else
246  {
247  iterativeSolver.compute(A1);
248  assert(iterativeSolver.info()==Eigen::Success && "SOLVER FAILED");
249  }
250 
251  model::cout<<" ["<<(std::chrono::duration<double>(std::chrono::system_clock::now()-t1)).count()<<" sec]"<<std::endl;
252  }
253 
254  // Solve
255  const auto t2= std::chrono::system_clock::now();
256  Eigen::VectorXd b1(T.transpose()*(b-A*g));
257  // Eigen::VectorXd x(Eigen::VectorXd::Zero(gSize));
258  Eigen::VectorXd x(Eigen::VectorXd::Zero(b1.rows()));
259 
260 
261  if(use_directSolver)
262  {
263 #ifdef _MODEL_PARDISO_SOLVER_
264  model::cout<<"PardisoLLT: solving..."<<std::flush;
265 #else
266  model::cout<<"SimplicialLLT: solving..."<<std::flush;
267 #endif
268  // x=T*directSolver.solve(b1)+g;
269  x=directSolver.solve(b1);
270  assert(directSolver.info()==Eigen::Success && "SOLVER FAILED");
271  const double b1Norm(b1.norm());
272  if(b1Norm>0.0)
273  {
274  const double axbNorm((A1*x-b1).norm());
275  if(axbNorm/b1Norm>tolerance)
276  {
277  model::cout<<"norm(A*x-b)/norm(b)="<<axbNorm/b1Norm<<std::endl;
278  model::cout<<"tolerance="<<tolerance<<std::endl;
279  assert(0 && "SOLVER FAILED");
280  }
281  }
282  }
283  else
284  {
285  model::cout<<"ConjugateGradient: solving..."<<std::flush;
286  iterativeSolver.setTolerance(tolerance);
287  // x=T*iterativeSolver.solveWithGuess(b1,guess)+g;
288  x=iterativeSolver.solveWithGuess(b1,guess);
289  model::cout<<" (relative error ="<<iterativeSolver.error()<<", tolerance="<<iterativeSolver.tolerance()<<")";
290  assert(iterativeSolver.info()==Eigen::Success && "SOLVER FAILED");
291  }
292 
293 
294 
295 
296  model::cout<<" ["<<(std::chrono::duration<double>(std::chrono::system_clock::now()-t2)).count()<<" sec]"<<std::endl;
297  return T*x+g;
298  }
