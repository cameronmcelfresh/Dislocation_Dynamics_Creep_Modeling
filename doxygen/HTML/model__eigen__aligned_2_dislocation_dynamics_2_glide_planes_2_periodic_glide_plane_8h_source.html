123  {

126 
127  if(link->sink->sID==this->sID)
128  {// edge ends at this node, so link is an inLink
129 
130  // Update _patchConnectivities
131  NodalConnectivityType& patchConnectivity(patchConnectivities()[link->patch->sID]);
132  assert(patchConnectivity.inEdge==nullptr);
133  patchConnectivity.inEdge=link;
134  if(patchConnectivity.outEdge)
135  {// and outEdge of the same patch is connected to this node
136  assert(patchConnectivity.outEdge->prev==nullptr);
137  assert(link->next==nullptr);
138  patchConnectivity.outEdge->prev=link;
139  link->next=patchConnectivity.outEdge;
140  }
141 
142  // Update _neighborConnectivities
143  NodalConnectivityType& neighborConnectivity(neighborConnectivities()[link->source->sID]);
144  assert(neighborConnectivity.inEdge==nullptr || neighborConnectivity.inEdge==link);
145  neighborConnectivity.inEdge=link;
146  if(neighborConnectivity.outEdge)
147  {// and outEdge of the same patch is connected to this node
148  assert(neighborConnectivity.outEdge->twin==nullptr || neighborConnectivity.outEdge->twin==link);
149  assert(link->twin==nullptr || link->twin==neighborConnectivity.outEdge);
150  neighborConnectivity.outEdge->twin=link;
151  link->twin=neighborConnectivity.outEdge;
152  }
153 
154  }
155  else if(link->source->sID==this->sID)
156  {// edge starts at this node, so link is an outLink
157 
158  // Update _patchConnectivities
159  NodalConnectivityType& patchConnectivity(patchConnectivities()[link->patch->sID]);
160  assert(patchConnectivity.outEdge==nullptr);
161  patchConnectivity.outEdge=link;
162  if(patchConnectivity.inEdge)
163  {
164  assert(patchConnectivity.inEdge->next==nullptr);
