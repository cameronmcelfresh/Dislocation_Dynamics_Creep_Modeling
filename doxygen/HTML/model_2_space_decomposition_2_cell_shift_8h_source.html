56  static typename CellShiftTraits<dim,neighborOrder>::MatrixType getShifts()
57  {
59  const typename CellShiftTraits<dim-1,neighborOrder>::MatrixType shiftsLower(CellShift<dim-1,neighborOrder>::getShifts());
60  const typename CellShiftTraits<1,neighborOrder>::MatrixType shifts1(CellShift<1,neighborOrder>::getShifts());
61  typename CellShiftTraits<dim,neighborOrder>::MatrixType temp;
62  for (int k=0;k<CellShift<1,neighborOrder>::Nneighbors;++k)
63  {
64  temp.template block<dim-1,CellShiftTraits<dim-1,neighborOrder>::Nneighbors>(0,k*CellShiftTraits<dim-1,neighborOrder>::Nneighbors)=shiftsLower;
65  temp.template block<1,CellShiftTraits<dim-1,neighborOrder>::Nneighbors>(dim-1,k*CellShiftTraits<dim-1,neighborOrder>::Nneighbors).setConstant(shifts1(k));
66  }
67  return temp;
68  }
