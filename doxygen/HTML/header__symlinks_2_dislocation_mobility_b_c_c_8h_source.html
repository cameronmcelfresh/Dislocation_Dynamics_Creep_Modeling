119  double velocity(const MatrixDim& S,
120  const VectorDim& b,
121  const VectorDim& xi,
122  const VectorDim& n,
123  const double& T,
124  const double& dL,
125  const double& dt,
126  const bool& use_stochasticForce) const
127  {
128 
129  const double bNorm=b.norm();
130  const VectorDim s = b/bNorm;
131  const VectorDim n1 = Eigen::AngleAxisd(M_PI/3.0,s)*n;
132 
133  // Compute components of non-Schmid model
134  const double tau=s.transpose()*S*n; // magnitude of resolved shear stress
135  const double tauOrt=n.cross(s).transpose()*S*n;
136  const double tau1=s.transpose()*S*n1; // resolved schear stress on
137  const double tauOrt1=n1.cross(s).transpose()*S*n1;
138 
139  const double num=std::fabs(tau+a1*tau1);
140  const double den=a0*tauC*sigmoid((a2*tauOrt+a3*tauOrt1)/a0/tauC);
141 
142  assert(den>0.0 && "den must be > 0.");
143 
144  const double Theta=num/den;
145  const double dg = (Theta<1.0)? (std::pow(1.0-std::pow(Theta,p),q)-T/T0) : 0.0;
146  const double dg1 = (dg>0.0)? dg : 0.0;
147  const double expCoeff = exp(-dH0*dg1/(2.0*kB_eV*T));
148 
149  // Compute screw drag coeff
150  const double sgm=0.5*sigmoid(-0.5*(0.05-dg1)/0.05);
151  const double Bs=Bk*w/(2.0*h)*(1.0-sgm)+(B0s+B1s*T)*sgm; //kink-dominated to drag-dominated interpolation
152 
153  // Compute screw velocity
154  double vs=std::fabs(tau)*bNorm/Bs*expCoeff;
155 
156  // Compute edge velocity
157  double ve=std::fabs(tau)*bNorm/(B0e+B1e*T);
158 
159  if(use_stochasticForce)
160  {
161  vs+=StochasticForceGenerator::velocity(kB,T,Bs,dL,dt);
162  ve+=StochasticForceGenerator::velocity(kB,T,B0e+B1e*T,dL,dt);
163  }
164 
165  // Interpolate ve and vs
166  const double cos2=std::pow(s.dot(xi),2);
167  return vs*cos2+ve*(1.0-cos2);
168  }
