412  bool depthFirstSearch (std::set<size_t>& searchedNodes, const size_t& ID, const size_t& N = ULONG_MAX) const
413  {
414  bool reached(this->sID==ID);
415  if (N!=0 && !reached)
416  {
417  assert(searchedNodes.insert(this->sID).second && "CANNOT INSERT CURRENT NODE IN SEARCHED NODES"); // this node has been searched
418  for (const auto& neighborIter : neighbors())
419  {
420  if (searchedNodes.find(std::get<0>(neighborIter.second)->sID)==searchedNodes.end())
421  { // neighbor not searched
422  reached=std::get<0>(neighborIter.second)->depthFirstSearch(searchedNodes,ID,N-1); // ask if neighbor can reach
423  if (reached)

425  break;
426  }
