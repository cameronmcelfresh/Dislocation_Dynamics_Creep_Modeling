30  static bool checkIntersection(const VectorDimD& C,
31  const VectorDimD& p1,
32  const VectorDimD& n1,
33  const VectorDimD& p2,
34  const VectorDimD& n2,
35  const double& tol)
36  {
37 
38  // Check that segments C-p1 and C-p2 are orthogonal to n1 and n2, respectively
39  const double Cp1((C-p1).norm());
40  const double Cp2((C-p2).norm());
41  const bool Cp1OK((Cp1<tol)? true : fabs((C-p1).dot(n1))<tol*Cp1);
42  const bool Cp2OK((Cp2<tol)? true : fabs((C-p2).dot(n2))<tol*Cp2);
43  const bool success(Cp1OK && Cp2OK);
44  if(!success)
45  {
46  model::cout<<"PlanePlaneIntersection FAILED"<<std::endl;
47  model::cout<<std::setprecision(15)<<std::scientific<<"C="<<C.transpose()<<std::endl;
48  model::cout<<std::setprecision(15)<<std::scientific<<"p1="<<p1.transpose()<<std::endl;

51  model::cout<<std::setprecision(15)<<std::scientific<<"n2="<<n2.transpose()<<std::endl;
52  model::cout<<std::setprecision(15)<<std::scientific<<"Cp1="<<Cp1<<std::endl;

54  model::cout<<std::setprecision(15)<<std::scientific<<"tol="<<tol<<std::endl;
55  }
56 
57  return success;

60  /**********************************************************************/
61  static SolutionType findIntersection(const VectorDimD& p1,
62  const VectorDimD& N1,
63  const VectorDimD& p2,
64  const VectorDimD& N2,
65  const double& tol)
66  {
67 
68  const double normN1(N1.norm());
69  const double normN2(N2.norm());
70  assert(normN1>tol);
71  assert(normN2>tol);
72 
73  const VectorDimD n1(N1/normN1);
74  const VectorDimD n2(N2/normN2);
75  const VectorDimD D(n1.cross(n2));
76  const double normD(D.norm());
77 
78  if(normD>tol)
79  {
80  // Find the Cartesian point P which minimizes (C-p1)^2+(C-p2)^2 under
81  // the constraints (C-p1)*n1=0 and (C-p2)*n2=0
82  Eigen::Matrix<double,dim+2,dim+2> M(Eigen::Matrix<double,dim+2,dim+2>::Zero());
83  M.template block<dim,dim>(0,0)=2.0*Eigen::Matrix<double,dim,dim>::Identity();
84  M.template block<dim,1>(0,dim+0)=n1;
85  M.template block<dim,1>(0,dim+1)=n2;
86  M.template block<1,dim>(dim+0,0)=n1.transpose();
87  M.template block<1,dim>(dim+1,0)=n2.transpose();
88 
89  Eigen::Matrix<double,dim+2,1> b(Eigen::Matrix<double,dim+2,1>::Zero());
90  b.template segment<dim>(0)=p1+p2;
91  b(dim+0)=p1.dot(n1);
92  b(dim+1)=p2.dot(n2);
93 
94  const VectorDimD C=M.ldlt().solve(b).template segment<dim>(0);
95  assert(checkIntersection(C,p1,n1,p2,n2,tol) && "PlanePlaneIntersection FAILED.");
96 
97  return std::make_tuple(INCIDENT,C,D/normD);
98  }
99  else
100  {
101  if(fabs((p1-p2).dot(n1))<tol)
102  {// coincident planes
103  return std::make_tuple(COINCIDENT,0.5*(p1+p2),VectorDimD::Zero());
104  }
105  else
106  {// paralle planes
107  return std::make_tuple(PARALLEL,0.5*(p1+p2),VectorDimD::Zero());
108  }
109  }
110  }
