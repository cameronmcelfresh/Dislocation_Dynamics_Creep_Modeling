125  template<typename LinkType>
126  static VectorDim getGlideVelocity(const LinkType& parentSegment,
127  const VectorDim& r,
128  const VectorDim& fPK,
129  const MatrixDim& S,
130  const VectorDim& rl,
131  const double& dL
132  )
133  {
134  VectorDim n(parentSegment.glidePlaneNormal()); // plane normal
135  VectorDim b(parentSegment.burgers()); // Burgers vector
136  VectorDim t(rl); // tangent vector
137 
138  // Select right-handed normal whenever possible
139  if(parentSegment.loopLinks().size()==1)
140  {// pick right-handed normal for n
141  const typename LinkType::LoopLinkType& loopLink(**parentSegment.loopLinks().begin());
142  if(std::fabs(loopLink.loop()->slippedArea())>FLT_EPSILON)
143  {
144  if(parentSegment.source->sID!=loopLink.source()->sID)
145  {// NetworkLink and LoopLink are oriented in opposite direction
146  b*=-1.0;
147  t*=-1.0;
148  }
149  }
150  }
151 
152  VectorDim glideForce = fPK-fPK.dot(n)*n;
153  double glideForceNorm(glideForce.norm());
154 
155  if(glideForceNorm<FLT_EPSILON && parentSegment.network().use_stochasticForce)
156  {
157  glideForce=parentSegment.chord().cross(n);
158  glideForceNorm=glideForce.norm();
159  if(glideForceNorm>FLT_EPSILON)
160  {
161  glideForce/=glideForceNorm;
162  }
163  }
164 
165  VectorDim vv=VectorDim::Zero();
166  if(glideForceNorm>FLT_EPSILON)
167  {
168  double v =parentSegment.network().poly.mobility->velocity(S,b,t,n,
169  parentSegment.network().poly.T,
170  dL,parentSegment.network().simulationParameters.dt,parentSegment.network().use_stochasticForce);
171  assert((parentSegment.network().use_stochasticForce || v>= 0.0) && "Velocity must be a positive scalar");
172  const bool useNonLinearVelocity=true;
173  if(useNonLinearVelocity && v>FLT_EPSILON)
174  {
175  v= 1.0-std::exp(-v);
176  }
177 
178  for(const auto& inclusion : parentSegment.network().eshelbyInclusions() )
179  {// Add EshelbyInclusions stress
180  if(inclusion.second.contains(r))
181  {
182  v*=inclusion.second.mobilityReduction;
183  }
184  }
185 
186 
187  vv= v * glideForce/glideForceNorm;
188  }
189  return vv;
190  }
