273  static double planarSolidAngle(const VectorDim& x,
274  const VectorDim& planePoint,
275  const VectorDim& rhN,
276  const std::vector<std::pair<VectorDim,VectorDim>>& polygonSegments)
277  {
278  double temp(0.0);
279  const double posNorm((x-planePoint).norm());
280  const double dotProd((x-planePoint).dot(rhN));
281  if(std::fabs(dotProd)>FLT_EPSILON*posNorm)
282  {// x is outside the plane of the loop
283  const VectorDim s(sgn(dotProd)*rhN); // s points along +n for points above, and along -n for points below
284  for(const auto& pair : polygonSegments)
285  {
286  VectorDim e1(pair.first-x);
287  const double e1Norm(e1.norm());
288  if(e1Norm>FLT_EPSILON)
289  {
290  e1/=e1Norm;
291  VectorDim Y1(pair.second-x);
292  const double Y1norm(Y1.norm());
293  if(Y1norm>FLT_EPSILON)
294  {
295  Y1/=Y1norm;
296  VectorDim e3(e1.cross(Y1));
297  const double e3Norm(e3.norm());
298  if(e3Norm>FLT_EPSILON)
299  {// e1 and Y1 are not align. If they are the projection on the unit sphere is a point and therefore there is no contribution to solid angle
300  e3/=e3Norm; // normalize e3
301  const VectorDim e2(e3.cross(e1));
302  const double ydy(e1.dot(Y1));
303  const double w=sqrt((1.0-ydy)/(1.0+ydy));
304  const double oneA2=sqrt(1.0+DislocationStress<dim>::a2);
305  const double s3(s.dot(e3));
306  const double s3A2=sqrt(std::pow(s3,2)+DislocationStress<dim>::a2);
307  temp+=2.0*s3/oneA2/s3A2*atan(s3A2*w/(oneA2-s.dot(e1)-s.dot(e2)*w));
308  }
309  }
310  }
311  }
312  }
313  return temp;
314  }

439  void updateGeometry()

441  nA.setZero();

443  if(glidePlane)
444  {// remove numerical errors by projecting along glidePlane->unitNormal
445 
446  if(this->links().size())
447  {
448  const VectorDim P0(this->links().begin()->second->source()->get_P());
449  for(const auto& loopLink : this->links())
450  {
451  nA+= 0.5*(loopLink.second->source()->get_P()-P0).cross(loopLink.second->sink()->get_P()-loopLink.second->source()->get_P());
452  }
453  }
454 
455 
456  // nA=nnDot*glidePlane->unitNormal;
457  // nA=nA.dot(glidePlane->n.cartesian().normalized())*glidePlane->n.cartesian().normalized();
458 
459  _slippedArea=nA.norm();
460  _rightHandedUnitNormal= _slippedArea>FLT_EPSILON? (nA/_slippedArea).eval() : VectorDim::Zero();
461 
462  const double nnDot(_rightHandedUnitNormal.dot(glidePlane->unitNormal));
463  _rightHandedNormal= nnDot>=0.0? glidePlane->n : ReciprocalLatticeDirection<dim>(glidePlane->n*(-1));
464  _rightHandedUnitNormal=_rightHandedNormal.cartesian().normalized();
465  }
