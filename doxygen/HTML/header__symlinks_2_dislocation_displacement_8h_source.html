173  template <typename ParticleType, typename CellContainerType>
174  static MatrixType multipole(const ParticleType& field,const CellContainerType& farCells)
175  {
180  MatrixType temp(MatrixType::Zero());
181  for(auto cell : farCells)
182  {
183 #ifdef _MODEL_ENABLE_CELL_VERTEX_ALPHA_TENSORS_
184  for(size_t v=0;v<Pow<2,dim>::value;++v)
185  {
186  VectorDim r(field.P-cell.second->vertices().col(v));
187  const double R(r.norm());
188  r/=R; // normalize R;
189  const MatrixDim& alpha(std::get<1>(*cell.second)[v]);
190  const VectorDim a(DislocationStress<dim>::axialVector(alpha));
191  temp += (+2.0*MaterialType::C1/(1.0+r.dot(field.S))*alpha*(field.S.cross(r))
192  /* */ - MaterialType::C3*a
193  /* */ - a.dot(r)*r)/R;
194  }
195 #else
196  VectorDim r(field.P-cell.second->center);
197  const double R(r.norm());
198  r/=R; // normalize R;
199  const MatrixDim& alpha(std::get<1>(*cell.second));
200  const VectorDim a(DislocationStress<dim>::axialVector(alpha));
201  temp += (+2.0*MaterialType::C1/(1.0+r.dot(field.S))*alpha*(field.S.cross(r))
202  /* */ - MaterialType::C3*a
203  /* */ - a.dot(r)*r)/R;
204 #endif
205  }
206  return MaterialType::C4*temp; // C4=1/(8*pi*(1-nu))
207  }
