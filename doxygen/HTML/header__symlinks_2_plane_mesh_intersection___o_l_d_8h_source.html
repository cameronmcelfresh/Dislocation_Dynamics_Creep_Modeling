56  static PlaneMeshIntersectionContainerType reducedPlaneMeshIntersection(const PlaneMeshIntersectionContainerType& mpi)
57  {
58  PlaneMeshIntersectionContainerType temp;
59 
60  size_t tailID=mpi.size()-1;
61  for(size_t k=0;k<mpi.size();++k)
62  {
63  const size_t k1=(k==mpi.size()-1? 0 : k+1);
64  const VectorDim dX=mpi[k].second-mpi[tailID].second;
65  const VectorDim dXnew=mpi[k1].second-mpi[k].second;
66  const double dXnorm=dX.norm();
67  const double dXnewNorm=dXnew.norm();
68  if( dXnorm > FLT_EPSILON
69  && dXnewNorm > FLT_EPSILON
70  && dX.cross(dXnew).norm()>FLT_EPSILON*dXnorm*dXnewNorm)
71  {// non zero angle
72  tailID=k;
73  temp.push_back(mpi[k]);
74  }
75 
76  }
77 // model::cout<<"("<<mpi.size()<<"->"<<temp.size()<<")"<<std::endl;

180  static SiblingsContainerType validUncles(const Simplex<dim,0>& vertex,
181  const int& rID,
182  const VectorDim& N)
183  {
184 
185  // Collect triangles attached to vertex
186  std::set<const Simplex<dim,dim-1>*> grandParents;
187  for(const auto& parent : vertex.parents())
188  {
189  for(const auto& grandParent : parent.second->parents())
190  {
191  grandParents.insert(grandParent.second);
192  }
193  }
194 
195  // Loop over triangles and collect valid edges
196  SiblingsContainerType uncles;
197  for(const auto& grandParent : grandParents)
198  {
199  if( (grandParent->isBoundarySimplex() || grandParent->isRegionBoundarySimplex())
200  && grandParent->isInRegion(rID)

348  /**********************************************************************/
349  static RootContainerType planeEdgeIntersection(const VectorDim& P0,
350  const VectorDim& n,
351  const Simplex<dim,dim-2>& edge,
352  std::set<const EdgeSimplexType*>& tested)
353  {
354  // std::deque<VectorDim> temp;
355  tested.insert(&edge);
356 
357  RootContainerType temp;
358 
359  const VectorDim& v0(edge.child(0).P0);
360  const VectorDim& v1(edge.child(1).P0);
361 
362  // check intersection of v0->v1 with plane
363  // x=v0+u(v1-v0)
364  // (x-P0).n=0
365  // (v0+u(v1-v0)-P0).n=0
366  // u=(P0-v0).n/(v1-v0).n;
367  const double edgeNorm=(v1-v0).norm();
368  assert(edgeNorm>FLT_EPSILON && "mesh edge has zero norm.");
369  const double den=(v1-v0).dot(n);
370  const double num=(P0-v0).dot(n);
371  const double P0v0norm=(P0-v0).norm();
372 
373  const double numCheck= (P0v0norm<FLT_EPSILON)? 0.0 : num/P0v0norm;
374 
375  if (fabs(den/edgeNorm)>meshIntersectionTol)
376  {
377  // edge intersects plane
378  const double u=num/den;
379 
380  if(fabs(u)<meshIntersectionTol)
381  {
382  temp.emplace_back(v0,&edge.child(0));
383  }
384  else if (u>=meshIntersectionTol && u<=1.0-meshIntersectionTol)
385  {
386  temp.emplace_back((1.0-u)*v0 + u*v1,nullptr);
387  }
388  else if (fabs(1.0-u)<meshIntersectionTol)
389  {
390  temp.emplace_back(v1,&edge.child(1));
391  }
392  else
393  {// no roots
394 
395  }
396 
397  }
398  else
399  {
400  if (fabs(numCheck)>meshIntersectionTol)
401  {// edge is parallel to plane, no intersection
402 
403  }
404  else
405  {// edge is coplanar
406  temp.emplace_back(v0,&edge.child(0));
407  temp.emplace_back(v1,&edge.child(1));
408  }
409  }
410 
411  return temp;
412  // return std::make_pair(&edge,temp);
413  }
