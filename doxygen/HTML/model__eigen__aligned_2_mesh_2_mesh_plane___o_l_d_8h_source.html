71  const int& rID,

73  {
78  //
79  // std::cout<<"getMeshBoundarySegments start"<<std::endl;
80  // Collect all intersection points
81  // Plane<dim> plane(p,n);
82  BoundingMeshSegments<dim> temp;
83  const MatrixDim R(plane.localRotationMatrix());
84 
85  for(const auto& face : mesh.region(rID)->faces())
86  {
87  //std::cout<<face.second->outNormal()<<std::endl;
88  PlanePlaneIntersection<dim> ppi(plane,face.second->asPlane());
89 
90  if(ppi.type==PlanePlaneIntersection<dim>::INCIDENT)
91  {// Collect all intersection points with lines defined by convexHull of face
92  //std::cout<<"incident"<<std::endl;
93  std::vector<VectorDim> pts;
94  //std::cout<<"face.second->convexHull().size()="<<face.second->convexHull().size()<<std::endl;
95 
96  for(size_t k=0;k<face.second->convexHull().size();++k)
97  {
98  const size_t k1(k==face.second->convexHull().size()-1? 0 : k+1);
99  PlaneSegmentIntersection<dim> psi(plane.P,plane.unitNormal,face.second->convexHull()[k]->P0,face.second->convexHull()[k1]->P0);
100 
101  if(psi.type==PlaneSegmentIntersection<dim>::INCIDENT)
102  {
103  pts.push_back(psi.x0);
104  }
105  else if(psi.type==PlaneSegmentIntersection<dim>::COINCIDENT)
106  {
107  pts.push_back(psi.x0);
108  pts.push_back(psi.x1);
109  break;
110  }
111  }
112 
113  // Compute ConvexHull of intersection points
114  ConvexHull<2,VectorDim> hull;
115  for(const auto& pt : pts)
116  {
117  VectorDim x(R*(pt-plane.P));
118 // HullPoint<2,VectorDim> hp({x[0],x[1]},&pt);
119 // hull.push_back(hp);
120  hull.emplace(std::array<double,2>{x[0],x[1]},&pt);
121  }
122 
123  const auto hullPts=hull.getPoints();
124  if(hullPts.size()==2)
125  {
126  //std::cout<<"a"<<std::endl;
127  temp.emplace_back(*hullPts[0].t,*hullPts[1].t,face.second.get());
128  }
129  else if(hullPts.size()>2)
130  {
131  //std::cout<<"b"<<std::endl;
132  for(size_t k=0;k<hullPts.size();++k)
133  {
134  const size_t k1(k==hullPts.size()-1? 0 : k+1);
135  temp.emplace_back(*hullPts[k].t,*hullPts[k1].t,face.second.get());
136  }
137  }
138  else
139  {// don't push points
140  //std::cout<<"c"<<std::endl;
141  }
142 
143  }
144  else if (ppi.type==PlanePlaneIntersection<dim>::COINCIDENT)
145  {
146  //std::cout<<"coincident"<<std::endl;
147  temp.clear();
148  for(size_t k=0;k<face.second->convexHull().size();++k)
149  {
150  const size_t k1(k==face.second->convexHull().size()-1? 0 : k+1);
151  temp.emplace_back(face.second->convexHull()[k]->P0,face.second->convexHull()[k1]->P0,face.second.get());
152  }
153  break;
154  }

162  //std::cout<<"unsorted hull"<<std::endl;
163  for(const auto& pt : temp)

165 
166  VectorDim x(R*(0.5*(pt.P0+pt.P1)-plane.P));
167  finalHull.emplace(std::array<double,2>{x[0],x[1]},&pt); // THE PROBLEM HERE IS THAT IF COINCIDENT POINTS FROM DIFFERENCE FACES EXIST, THEN ONLY ONE OF THEM IS KEPT. E.G. A PLANE CUTTING AT THE INTERSECTION OF TWO FACES. IF WE HAD UNIQUE FACE EDGES WITH POINTERS TO THE ADJECENT FACES WE COULD SOLVE THIS

169  }

172  assert(hullPts.size()==temp.size());

175 // for(const auto& seg : hullPts)

180  for(size_t k=0;k<hullPts.size();++k)
181  {
182  const auto& seg(*hullPts[k].t);
183  if(k==0)
184  {
185  const auto& seg1(*hullPts[1].t);
186 
187  if((seg.P1-seg1.P0).norm()<FLT_EPSILON || (seg.P1-seg1.P1).norm()<FLT_EPSILON)
188  {// first segment is oriented in the right direction
189  sortedTemp.push_back(seg);
190  }
191  else
192  {
193  sortedTemp.emplace_back(seg.P1,seg.P0,seg.faces);
194  }
195  }
196  else
197  {
198  if((sortedTemp.back().P1-seg.P0).norm()<FLT_EPSILON)
199  {
200  sortedTemp.push_back(seg);
201  }
202  else if((sortedTemp.back().P1-seg.P1).norm()<FLT_EPSILON)
203  {
204  sortedTemp.emplace_back(seg.P1,seg.P0,seg.faces);
205  }
206  else
207  {
208  //std::cout<<"k="<<k<<" of "<<hullPts.size()<<std::endl;
209  assert(false && "DISCONNECTED FACE BOUNDARY");
210  }
211  }
212  //std::cout<<sortedTemp.back().P0.transpose()<<","<<sortedTemp.back().P1.transpose()<<std::endl;
213 
214  }
215  assert((sortedTemp.back().P1-sortedTemp.front().P0).norm()<FLT_EPSILON && "OPEN FACE BOUNDARY");
216 
217  assert(sortedTemp.size()==temp.size());
218 
219 
220  return sortedTemp;
