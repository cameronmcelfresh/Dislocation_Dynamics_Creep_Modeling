58  bool isValidJunction((bndJunction || gbndJunction) && DN.simulationParameters.simulationType!=2);
59  if( !isValidJunction

61  && !linkB->isBoundarySegment()
62  && !linkA->isGrainBoundarySegment()
63  && !linkB->isGrainBoundarySegment())
64  {// Check force condition for internal segments
65 
66  const VectorDim chordA(linkA->sink->get_P()-linkA->source->get_P());
67  const double LA(chordA.norm());
68  const VectorDim chordB(linkB->sink->get_P()-linkB->source->get_P());
69  const double LB(chordB.norm());
70 
71  if(LA>FLT_EPSILON && LB>FLT_EPSILON)
72  {
73  StressStraight<dim> stressA(ssd.x0-infiniteLineLength/LA*chordA,
74  ssd.x0+infiniteLineLength/LA*chordA,
75  linkA->burgers());

78  ssd.x1+infiniteLineLength/LB*chordB,

82  const VectorDim forceOnB=(stressA.stress(ssd.x1)*linkB->burgers()).cross(chordB);
