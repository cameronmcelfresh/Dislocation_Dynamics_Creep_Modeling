253  /**********************************************************************/
254  void assembleGlide()
255  {
256  this->updateForcesAndVelocities(*this,quadPerLength,false);
257  Fq= this->quadraturePoints().size()? this->nodalVelocityVector() : VectorNdof::Zero();
258  Kqq=this->nodalVelocityMatrix(*this);
259  h2posMap=this->hermite2posMap();
260  Mseg.setZero(Ncoeff*dim,h2posMap.size()*dim);
261  size_t c=0;
262  for(const auto& pair : h2posMap)
263  {
264  for(int r=0;r<Ncoeff;++r)
265  {
266  Mseg.template block<dim,dim>(r*dim,c*dim)=pair.second.first(r)*MatrixDim::Identity();
267  }
268  c++;
269  }
270  }

272  /**********************************************************************/
273  void addToGlobalAssembly(std::deque<Eigen::Triplet<double> >& kqqT,
274  Eigen::VectorXd& FQ) const
275  {
278  if(!hasZeroBurgers())
279  {
280  const Eigen::MatrixXd tempKqq(Mseg.transpose()*Kqq*Mseg); // Create the temporaty stiffness matrix and push into triplets
281  size_t localI=0;
282  for(const auto& pairI : h2posMap)
283  {
284  for(int dI=0;dI<dim;++dI)
285  {
286  const size_t globalI=pairI.first*dim+dI;
287  size_t localJ=0;
288  for(const auto& pairJ : h2posMap)
289  {
290  for(int dJ=0;dJ<dim;++dJ)
291  {
292  const size_t globalJ=pairJ.first*dim+dJ;
293 
294  if (std::fabs(tempKqq(localI,localJ))>FLT_EPSILON)
295  {
296  kqqT.emplace_back(globalI,globalJ,tempKqq(localI,localJ));
297  }
298  localJ++;
299  }
300  }
301  localI++;
302  }
303  }
304 
305  const Eigen::VectorXd tempFq(Mseg.transpose()*Fq); // Create temporary force vector and add to global FQ
306  localI=0;
307  for(const auto& pairI : h2posMap)
308  {
309  for(int dI=0;dI<dim;++dI)
310  {
311  const size_t globalI=pairI.first*dim+dI;
312 
313  FQ(globalI)+=tempFq(localI);
314 
315  localI++;
316  }
317  }
318  }
319  }
