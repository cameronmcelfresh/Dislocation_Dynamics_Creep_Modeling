31  static std::deque<Eigen::Matrix<double,dim,1>> reducePermutations(const Eigen::Matrix<double,dim,Eigen::Dynamic>& m)
32  {
33  std::deque<Eigen::Matrix<double,dim,1>> temp;
34 
35  for (int c=0;c<m.cols();++c)
36  {
37  bool isUnique=true;
38  for(const auto& v : temp)
39  {
40  isUnique*=( (m.col(c)-v).norm()>FLT_EPSILON && (m.col(c)+v).norm()>FLT_EPSILON);
41 
42  if(!isUnique)
43  {
44  break;
45  }
46  }
47 
48  if(isUnique)
49  {
50  temp.push_back(m.col(c));
51  }
52  }
