279  std::pair<double,std::pair<double,VectorDim> > closestPoint(const VectorDim& P0) const
280  {
288  // solve (P-P0)*dP/du=0
289 
290  // The polynomial coefficients of this spine segment
291  Eigen::Matrix<double,dim,Ncoeff> coeffs(this->polynomialCoeff());
292  coeffs.col(0)-=P0;
293 
294 
295  // The derivative of the polynomial coefficients
296  Eigen::Matrix<double,dim,Ncoeff-1> dcoeffs(Eigen::Matrix<double,dim,Ncoeff-1>::Zero());
297  for (int i=0;i<Ncoeff-1;++i)
298  {
299  dcoeffs.col(i)=(i+1)*coeffs.col(i+1);
300  }
301 
302  Eigen::Matrix<double,1,2*Ncoeff-2> pcoeffs(Eigen::Matrix<double,1,2*Ncoeff-2>::Zero()); // degree of product = pOrder+(pOrder-1)=2*pOrder-1. nCoeffs of product = 2*pOrder-1+1= 2*pOrder = 2*Ncoeff-2
303 
304  // The polynomial coefficients of (P-P0)*dP/du, in reverse order
305  for (int i=0;i<Ncoeff;++i)
306  {
307  for (int j=0;j<Ncoeff-1;++j)
308  {
309  pcoeffs(2*Ncoeff-3-i-j) += coeffs.col(i).dot(dcoeffs.col(j));
310  }
311  }
312 
313  // Compute roots using the eigenvalue method
314  MatrixCompanion<2*Ncoeff-3> mc(pcoeffs);
315 
316  // sort roots according to distance to P0
317  std::map<double,std::pair<double,VectorDim>> rootMap;
318 
319  // for (int k=0;k<2*Ncoeff-3;++k)
320  for (size_t k=0;k<mc.rootSize;++k)
321  {
322  if (std::fabs(mc.root(k).imag())<FLT_EPSILON && mc.root(k).real()>0.0 && mc.root(k).real()<1.0 )
323  {
324 
325  VectorDim P(this->get_r(mc.root(k).real()));
326  rootMap.insert(std::make_pair((P-P0).norm(), std::make_pair(mc.root(k).real(),P) ));
327 
328  }
329 
330  }
331 
332  // check distance to limits of interval
333  rootMap.insert(std::make_pair((this->source->get_P()-P0).norm(), std::make_pair(0.0,this->source->get_P()) ));
334  rootMap.insert(std::make_pair((this-> sink->get_P()-P0).norm(), std::make_pair(1.0,this-> sink->get_P()) ));
335 
336  return *rootMap.begin();
337 
338  }
