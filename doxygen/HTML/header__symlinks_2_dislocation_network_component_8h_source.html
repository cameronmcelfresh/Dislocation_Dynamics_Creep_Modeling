287  void lumpedSolve(const size_t& runID)
288  {
289  TripletContainerType kqqT; // the vector of Eigen::Triplets corresponding to the matrix Kqq
290  Eigen::VectorXd Fq; // the vector of nodal forces
291  const size_t Ndof=assembleNCtriplets(kqqT,Fq);
292 
293  // Zienkiewicz (See [1], section 16.2.4) discusses three methods for lumping the mass matrix
294  TripletContainerType lumpedTriplets;
295  for(const auto& t : kqqT)
296  {
297  if(t.col()==t.row())
298  {
299  lumpedTriplets.push_back(t);
300  }
301  else
302  {
303  lumpedTriplets.emplace_back(t.col(),t.col(),0.5*t.value());
304  lumpedTriplets.emplace_back(t.row(),t.row(),0.5*t.value());
305  }
306  }
307 
308  SparseMatrixType kqq(Ndof,Ndof);
309  kqq.setFromTriplets(lumpedTriplets.begin(),lumpedTriplets.end());
310  Eigen::VectorXd Kd(kqq.diagonal());
311  Eigen::VectorXd x(Eigen::VectorXd::Zero(Ndof));
312 
313  if(outputKF)
314  {
315  assert(0 && "RE-ENABLE OUTPUT");
316  std::ofstream fileK("K_"+std::to_string(runID)+"_"+std::to_string(NC.sID)+".txt");
317 // fileK<<Kd;
318  std::ofstream fileF("F_"+std::to_string(runID)+"_"+std::to_string(NC.sID)+".txt");
319 // fileF<<Fq;
320  }
321 
322  // Check diagonal and force
323  for (int k=0;k<Kd.size();++k)
324  {
325  if(fabs(Kd(k))>FLT_EPSILON)
326  {// stiffness not zero
327  x(k)=Fq(k)/Kd(k);
328  }
329  else
330  {// stiffness is zero
331  assert(fabs(Fq(k))<FLT_EPSILON && "if stiffness is zero also force must be zero.");
332  }
333  }
334  storeNodeSolution(x.segment(0,Ndof));
335  }

338  void iterativeSolve()
339  {
340  // if(solvable)
341  // {
342  TripletContainerType kqqT; // the vector of Eigen::Triplets corresponding to the matrix Kqq
343  Eigen::VectorXd Fq; // the vector of nodal forces
344  const size_t Ndof=assembleNCtriplets(kqqT,Fq);
345 
346  // Find guess solution using current velocities
347  Eigen::VectorXd x0(Ndof);
348  size_t k=0;
349  for (typename NodeContainerType::iterator nodeIter=NC.nodeBegin();nodeIter!=NC.nodeEnd();++nodeIter){
350  x0.segment(NdofXnode*k,NdofXnode)=nodeIter->second->get_V();
351  ++k;
352  }
353 
354  bool usePreSolver=true;
355  if (usePreSolver)
356  {
357  // kqq is SPD, find the unconstrained solution with ConjugateGradient (faster)
358  SparseMatrixType kqq(Ndof,Ndof);
359  kqq.setFromTriplets(kqqT.begin(),kqqT.end());
360  Eigen::ConjugateGradient<SparseMatrixType> cg(kqq);
361 
362  if(cg.maxIterations()<10000)
363  {
364  cg.setMaxIterations(10000);
365  }
366  x0=cg.solveWithGuess(Fq,x0);
367  }
368 
369 
370  // Assemble constraints
371  size_t KPQ_row=Ndof; // rows 0-(Ndof-1) are taken by Kqq. Start placing constraints at row Ndof
372  assembleConstraints(kqqT,KPQ_row,true); // kqqT and KPQ_row are overwritten
373  SparseMatrixType KQQ(KPQ_row,KPQ_row);
374  KQQ.setFromTriplets(kqqT.begin(),kqqT.end()); // now KQQ is idefinite (NOT SPD) therefore conjugate-gradient cannot be used
375 
376  Eigen::VectorXd F(Eigen::VectorXd::Zero(KPQ_row));
377  F.segment(0,Ndof)=Fq;
378  Eigen::VectorXd x1(Eigen::VectorXd::Zero(KPQ_row));
379  x1.segment(0,Ndof)=x0;
380 
381  bool useMINRES=true;
382  if (useMINRES)
383  {
384  Eigen::MINRES<SparseMatrixType> solver(KQQ);
385  solver.setTolerance(FLT_EPSILON);
386  solver.setMaxIterations(10*F.rows());
387  x1=solver.solveWithGuess(F,x1);
388  if(solver.info()==Eigen::Success)
389  {
390  storeNodeSolution(x1.segment(0,Ndof));
391  }
392  else
393  {
394  std::cout<<"Solver did not converge\n";
395  std::cout<<"iterations="<<solver.iterations()<<"\n";
396  std::cout<<"maxIterations="<<solver.maxIterations()<<"\n";
397  std::cout<<"error="<<solver.error()<<"\n";
398  std::cout<<"tolerance="<<solver.tolerance()<<"\n";
399  assert(0 && "SOLVER DID NOT CONVERGE.");
400  }
401  }
402  else
403  {
404  Eigen::BiCGSTAB<SparseMatrixType,Eigen::IdentityPreconditioner> bcs(KQQ);
405  if(bcs.maxIterations()<10000)
406  {
407  bcs.setMaxIterations(10000);
408  }
409  x1=bcs.solveWithGuess(F,x1);
410  if(bcs.info()==Eigen::Success)
411  {
412  storeNodeSolution(x1.segment(0,Ndof));
413 
414  }
415  else
416  {
417  std::cout<<"Solver did not converge\n";
418  std::cout<<"iterations="<<bcs.iterations()<<"\n";
419  std::cout<<"maxIterations="<<bcs.maxIterations()<<"\n";
420  std::cout<<"error="<<bcs.error()<<"\n";
421  std::cout<<"tolerance="<<bcs.tolerance()<<"\n";
422  assert(0 && "SOLVER DID NOT CONVERGE.");
423  }
424  }
