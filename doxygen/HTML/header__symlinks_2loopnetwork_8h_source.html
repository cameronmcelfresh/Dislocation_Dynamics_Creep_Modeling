125  void connect(const SharedNodePtrType& n0,
126  const SharedNodePtrType& n1,
127  const std::shared_ptr<LoopType>& tempLoop)
128  {
129  VerboseLoopNetwork(1,"connecting "<<n0->sID<<"->"<<n1->sID<<" ("<<tempLoop->sID<<")"<<std::endl);
130  // assert(n0->sID!=n1->sID && "Cannot connect a node to itself");
131 
132  if(n0->sID!=n1->sID)
133  {
134  //check if n1->n0 exist for the same loop
135  // const auto iterPair = loopLinks().equal_range(std::pair<size_t,size_t>(n1->sID,n0->sID));
136  const auto key=LoopLinkType::networkLinkKey(n0,n1); // even for LoopLinks we store with key (minID,maxID)
137  const auto iterPair = loopLinks().equal_range(key);
138 
139  typename LoopLinkContainerType::const_iterator loopIter=iterPair.second;
140  for (loopIter=iterPair.first;loopIter!=iterPair.second;++loopIter)
141  {
142  if(loopIter->second.loop().get()==tempLoop.get() && // link in same loop
143  loopIter->second.source()->sID==n1->sID && loopIter->second.sink()->sID==n0->sID) // opposite link exists
144  {
145  break;
146  }
147 
148  }
149 
150  if(loopIter!=iterPair.second)
151  {// opposite link exists, disconnect it
152  loopLinks().erase(loopIter);
153  }
154  else
155  {// opposite link does not exists, connect n0->n1
156  loopLinks().emplace(std::piecewise_construct,
157  std::make_tuple(key.first,key.second),
158  std::make_tuple(n0,n1, tempLoop)
159  );
160  }
161  }

200  loopIter->second.source()->sID==n0->sID && loopIter->second.sink()->sID==n1->sID)

202  loopLinks().erase(loopIter);

746  bool contractSecond(const SharedNodePtrType& nA,const SharedNodePtrType& nB)
