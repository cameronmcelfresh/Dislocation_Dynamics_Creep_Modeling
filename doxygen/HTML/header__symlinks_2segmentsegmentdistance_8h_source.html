176  /**********************************************************************/
177  IntersectionContainerType intersectionSegment() const
178  {/*\returns the point(s) of intersection between the segments, if any.
179  * The return type is deque<tuple<x,t,u>>, where x is the intersection
180  * point, t the parameter on the first segment, u the parameter on the
181  * second segment. The deque is empty if no intersection was found.
182  * If one intersection is found, the deque contains that intersection.
183  * In case of coincident segments, the deque contains two points. These
184  * points define the line element over which the original segments overlap.
185  */
186 
187  IntersectionContainerType temp;
188 
189  if(dMin<FLT_EPSILON)
190  {// an intersection exists
191 
192  if(D1<tol || D2<tol)
193  {// At least one segment is degenerate
194  temp.emplace_back(0.5*(x0+x1),t,u);
195  }
196  else
197  {// both segments are not degenerate
198 
199 // std::cout<<"den="<<den<<std::endl;
200 // std::cout<<"tol="<<tol<<std::endl;
201 
202  if(fabs(den)>tol*D1*D2)
203  {// Step 1d: skew segments
204  temp.emplace_back(0.5*(x0+x1),t,u);
205  }
206  else
207  {// coincident segments. Since there must be an intersection, the two innermost points are the overlapping segment
208 
209  const double tC=(C-A).dot(d1)/D1; // value of t for point C
210  const double tD=(D-A).dot(d1)/D1; // value of t for point D
211 
212  std::multimap<double,VectorDim> ms;
213  ms.emplace(0.0,A);
214  ms.emplace(1.0,B);
215  ms.emplace(tC,C);
216  ms.emplace(tD,D);
217 
218  auto iter1=ms.begin();
219  std::advance(iter1,1);
220  auto iter2=ms.begin();
221  std::advance(iter2,2);
222 
223  if((iter1->second-iter2->second).norm()>FLT_EPSILON)
224  {
225  temp.emplace_back(iter1->second,iter1->first,(iter1->second-C).dot(d2)/D2);
226  temp.emplace_back(iter2->second,iter2->first,(iter2->second-C).dot(d2)/D2);
227  }
228  else
229  {
230  temp.emplace_back(0.5*(iter1->second+iter2->second),
231  0.5*(iter1->first+iter2->first),
232  0.5*((iter1->second-C).dot(d2)/D2+(iter2->second-C).dot(d2)/D2)
233  );
234 
235  }
236  }
237  }
238 
239  }
240 
241  return temp;
242  }
