58  template <typename FieldType,typename... OtherSourceTypes>
59  void computeNeighborField(const OtherSourceTypes&... otherSources)
60  {
61  // Nearest-neighbor interaction
63 #ifdef _OPENMP
64 #pragma omp parallel for
65 #endif
66  for (unsigned int k=0; k<this->size();++k)
67  {
68  if(static_cast<FieldPointBase<ParticleType,FieldType>* const>(&this->operator[](k))->enabled)
69  {
71  for (typename SpatialCellType::CellMapType::const_iterator cIter =this->operator[](k).neighborCellsBegin();
72  /* */ cIter!=this->operator[](k).neighborCellsEnd();
73  /* */ ++cIter)
74  {
76  for(typename SpatialCellType::ParticleContainerType::const_iterator qIter =cIter->second->particleBegin();
77  /* */ qIter!=cIter->second->particleEnd();
78  /* */ ++qIter)
79  {
80  if(static_cast<const SingleSourcePoint<_ParticleType,FieldType>* const>(*qIter)->enabled)
81  {
82  *static_cast<FieldPointBase<ParticleType,FieldType>* const>(&this->operator[](k)) += FieldType::compute(**qIter,this->operator[](k));
83  }
84  }
85  }
86 
87  // Non-nearest-neighbor interaction
88  if(FieldType::use_multipole)
89  {
90  typename SpatialCellType::CellMapType farCells(this->operator[](k).template farCells<_ParticleType>());
91  *static_cast<FieldPointBase<ParticleType,FieldType>* const>(&this->operator[](k)) += FieldType::multipole(this->operator[](k),farCells);
92  }
93 
94  // Add contribution of other sources
95  if(sizeof...(OtherSourceTypes))
96  {
97  *static_cast<FieldPointBase<ParticleType,FieldType>* const>(&this->operator[](k)) += FieldType::addSourceContribution(this->operator[](k),otherSources...);
98  }
99  }
100  }
101  }

107  void computeField(std::deque<OtherParticleType>& fpDeq, const OtherSourceTypes&... otherSources) const
108  {
109 #ifdef _OPENMP
110 #pragma omp parallel for

113  {

121  /* */ cIter!=neighborCells.end();
122  /* */ ++cIter)

126  /* */ qIter!=cIter->second->particleEnd();

129  //fpDeq[k].template field<FieldType>() += FieldType::compute(**qIter,fpDeq[k]);

131  {// source is enabled
132  *static_cast<FieldPointBase<OtherParticleType,FieldType>* const>(&fpDeq[k]) += FieldType::compute(**qIter,fpDeq[k]);
133  }
134  }
135  }
136 
137  // Non-nearest-neighbor interaction
138  if(FieldType::use_multipole)
139  {
140  typename SpatialCellType::CellMapType farCells(fpDeq[k].template farCells<_ParticleType>());

144  }
