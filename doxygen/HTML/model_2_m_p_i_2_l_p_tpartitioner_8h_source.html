66  // Resize

68  BaseWeightVectorType::resize(nProc,0.0);
69 
70  // Traverse BaseMapType in descending order (highest to lowest weight)
71  for (typename BaseMapType::const_reverse_iterator rIter =BaseMapType::rbegin();
72  /* */ rIter!=BaseMapType::rend();
73  /* */ ++rIter)
74  {
75  // Compute the index of minimum weight in BaseWeightVectorType
76  const size_t argMin(min_element(BaseWeightVectorType::begin(), BaseWeightVectorType::end())-BaseWeightVectorType::begin());
77  // Add the current weigth to the bin of minimum weigth
78  BaseWeightVectorType::operator[](argMin)+=rIter->first;
79  // Add the current pointer to the bin of minimum weight
80  BaseDequeVectorType::operator[](argMin).push_back(rIter->second);
81  }
82 
83  partitionIsValid=true;
