106  static MatrixType multipole(const ParticleType& field,const CellContainerType& farCells)
107  {
190  MatrixType temp(MatrixType::Zero());
191  for(auto cell : farCells)
192  {
193 #ifdef _MODEL_ENABLE_CELL_VERTEX_ALPHA_TENSORS_
194  for(size_t v=0;v<Pow<2,dim>::value;++v)
195  {
196  VectorDim R(field.P-cell.second->vertices().col(v));
197  const double R2(R.squaredNorm());
198  R/=sqrt(R2); // normalize R;
199  const MatrixDim& alpha(std::get<0>(*cell.second)[v]);
200  const VectorDim a(axialVector(alpha));
201  const MatrixDim S(skewMatrix(R));
202  temp += (MaterialType::C1*S*alpha-a*R.transpose()+0.5*R.dot(a)*(3.0*R*R.transpose()+I))/R2;
203 
204  }
205 #else
206  VectorDim R(field.P-cell.second->center);
207  const double R2(R.squaredNorm());
208  R/=sqrt(R2); // normalize R;
209  const MatrixDim& alpha(std::get<0>(*cell.second));
210  const VectorDim a(axialVector(alpha));
211  const MatrixDim S(skewMatrix(R));
212  temp += (MaterialType::C1*S*alpha-a*R.transpose()+0.5*R.dot(a)*(3.0*R*R.transpose()+I))/R2;
213 #endif
