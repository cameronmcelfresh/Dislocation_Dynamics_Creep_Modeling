49  std::pair<bool,double> mustBeContracted(const LinkType& segment) const
50  {
51  const double vTolcont=0.0;
52 
53  std::pair<bool,double> temp=std::make_pair(false,0.0);
54  const VectorDimD chord(segment.chord()); // this is sink->get_P() - source->get_P()
55  const double chordLength(chord.norm());
56  const VectorDimD dv(segment.sink->get_V()-segment.source->get_V());
57  bool endsAreApproaching( chord.dot(dv) < 0.0 );
58  double currentLmin=Lmin;
59  const int sourceNonZeroSize(segment.source->nonZeroNeighbors().size());
60  const int sinkNonZeroSize(segment.sink->nonZeroNeighbors().size());
61  if((sourceNonZeroSize>2 && sinkNonZeroSize==2) || (sourceNonZeroSize==2 && sinkNonZeroSize>2))
62  {
63  currentLmin=1.0;
64  }
65 
66 
67 
68  if (((endsAreApproaching || segment.isBoundarySegment())// ends are approaching
69  && dv.norm()*DN.simulationParameters.dt>vTolcont*chordLength // contraction is large enough compared to segment length
70  && chordLength<currentLmin // segment is small
71  )
72  // || segment.isSimpleBndSegment()
73  //|| segment.isSimpleSessile()
74  )
75  {
76  temp=std::make_pair(true,chordLength);
77  }
78  return temp;
79  }
