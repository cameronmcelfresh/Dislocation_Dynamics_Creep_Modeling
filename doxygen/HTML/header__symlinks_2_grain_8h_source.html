94  const ReciprocalLatticeDirectionType& N) const
95  {
96  std::deque<const LatticePlaneBase*> temp;

98  {
99  for (const auto& planeNormal : this->planeNormals())
100  {
101  if( B.dot(planeNormal)==0 && N.cross(planeNormal).squaredNorm()>0)
102  {
103  temp.push_back(&planeNormal);
104  }
105  }
106  }
