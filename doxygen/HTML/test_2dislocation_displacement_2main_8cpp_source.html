39 Eigen::Matrix<double,3,1> displacementStraightEdge(const Eigen::Matrix<double,3,1>& P)
40 {
45  Eigen::Matrix<double,3,1> disp(Eigen::Matrix<double,3,1>::Zero()); // 3x3 matrix of zeros
46 
47  const double& x(P(0));
48  const double& y(P(1));
49  const double x2(x*x);
50  const double y2(y*y);
51  const double R2(x2+y2);
52  const double& nu(Material<Isotropic>::nu);
53 
54  double at2yx=atan2(y,x); // atan2 output results in [-pi;pi]
55  if(at2yx<0.0)
56  {// transform at2yx in [0;2*pi]
57  at2yx+=2.0*M_PI;
58  }
59  at2yx-=M_PI; // divide equally jump across slip plane
60 
61  // From Hirth and Lothe, Therory of Dislocations 2nd Ed, p 78. Eq. (3-45) and (3.56)
62  // disp(0)= at2yx + x*y/(2.0*(1.0-nu)*R2);
63  // disp(1)= -((1.0-2.0*nu)/(4.0*(1.0-nu))*log(R2)+(x2-y2)/(4.0*(1.0-nu)*R2));
64 
65  // Eshelby (Brit J Appl Phys 1966 v.17)
66  disp(0)= at2yx - x*y/(2.0*(1.0-nu)*R2);
67  disp(1)= -(1.0-2.0*nu)/(4.0*(1.0-nu))*log(R2)+y2/(2.0*(1.0-nu)*R2);
68  // disp(1)= -(1.0-2.0*nu)/(4.0*(1.0-nu))*log(x2+y2)+(x2-y2)/(4.0*(1.0-nu)*R2);
69  disp *= 1.0/(2.0*M_PI);
70 
71  //From Nabarro, p.57ttttt
72  return disp;
73 }
