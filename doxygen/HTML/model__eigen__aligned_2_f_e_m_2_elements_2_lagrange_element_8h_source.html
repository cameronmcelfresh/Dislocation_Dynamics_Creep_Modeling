76  static Eigen::Matrix<double,1,Eigen::Dynamic> shapeFunctionCoeff(const int& N)
77  {
82  Eigen::Matrix<int,1,Eigen::Dynamic> prod(1,1);
83  prod<<1; // initialize the polynomial
84  int f(1); // initialize the factorial
85 
86  for (int n=0;n<N;++n)
87  {
88  Eigen::Matrix<int,1,2> temp2(-n,degree);
89  Eigen::Matrix<int,1,Eigen::Dynamic> temp1(1,prod.cols()+1);
90  temp1.setZero();
91  for (int i = 0;i < prod.cols();++i)
92  {
93  for (int j = 0;j < temp2.cols();++j)
94  {
95  temp1(i + j) += prod(i) * temp2(j); // convolution
96  }
97  }
98  prod=temp1; // overwrite product
99  if (n>0)
100  {
101  f*=n; // compute factorial
102  }
103  }
104  if (N>0)
105  {
106  f*=N; // add last term to factorial
107  }
108  return prod.cast<double>()/f;
109  }

139  static Eigen::Matrix<double,1,dim+1> sfcTimesBary(const int& n, const Eigen::Matrix<double,1,dim+1>& bary)
140  {
141  Eigen::Matrix<double,1,dim+1> temp(Eigen::Matrix<double,1,dim+1>::Zero());
142  for(int d=0;d<dim+1;++d) // this loop can be parallelized
143  {
144  Eigen::Matrix<double,degree+1,1> bPow(Eigen::Matrix<double,degree+1,1>::Ones());
145  for (int p=1;p<degree+1;++p) // start at 1 since bPow(0)=1 from initialization
146  {
147  // bPow(p)=std::pow(bary(d),p);
148  bPow(p)=bPow(p-1)*bary(d);
149  }
150  temp(d)=sfCoeffsVector[n].row(d).dot(bPow);
151  }
152  return temp;

160  {

163  for (int p=2;p<degree+1;++p) // start at 1 since bPow(0)=0 (unused) and bPow(1)=1 from initialization
