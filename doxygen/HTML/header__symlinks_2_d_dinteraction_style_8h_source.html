186  bool frameLoaded=false;

207  meshActor.update(frameID/*,lastFrameID,degPerStep,spinAxis*/);

213 
214 
215  const double spinAxisNorm(spinAxis.norm());
216  if(autoSpin && degPerStep && spinAxisNorm)
217  {
218  spinAxis/=spinAxisNorm;

366  virtual void OnLeftButtonDown()
367  {
368 // autoSpin=false;
369 
370  int* clickPos = this->GetInteractor()->GetEventPosition();
371 
372  // Pick from this location.
373  vtkSmartPointer<vtkPropPicker> picker =
374  vtkSmartPointer<vtkPropPicker>::New();
375  picker->Pick(clickPos[0], clickPos[1], 0, this->GetDefaultRenderer());
376 
377  // If we picked something before, reset its property
378  if (this->LastPickedActor)
379  {
380  this->LastPickedActor->GetProperty()->DeepCopy(this->LastPickedProperty);
381  }
382  this->LastPickedActor = picker->GetActor();
383  if (this->LastPickedActor)
384  {
385  // Save the property of the picked actor so that we can
386  // restore it next time
387  this->LastPickedProperty->DeepCopy(this->LastPickedActor->GetProperty());
388  // Highlight the picked actor by changing its properties
389  this->LastPickedActor->GetProperty()->SetColor(1.0, 0.0, 0.0);
390  this->LastPickedActor->GetProperty()->SetDiffuse(1.0);
391  this->LastPickedActor->GetProperty()->SetSpecular(0.0);
392  }
393 
394  // Forward events
395  vtkInteractorStyleTrackballCamera::OnLeftButtonDown();
396  }
