83  /**********************************************************************/
84  static MatrixHigherDimD get_L2X()
85  {/*!/returns the barycentric-to-standard coordinate transformation matrix.
86  *
87  * For example, in 2d, the transformation is:
88  * \f[
89  * \left[\begin{array}{lll}
90  * v_{11}&v_{21}&v_{31}\\
91  * v_{12}&v_{22}&v_{32}\\
92  * 1&1&1\\
93  * \end{array}
94  * \right]
95  * \left[\begin{array}{l}
96  * \lambda_1\\
97  * \lambda_2\\
98  * \lambda_3\\
99  * \end{array}
100  * \right]=
101  * \left[\begin{array}{c}
102  * s_1\\
103  * s_2\\
104  * 1\\
105  * \end{array}
106  * \right]
107  * \f]
108  *
109  * In the sandard domain, we now adopt the convention that, for i<n,
110  * the i-th vertex is on the i-th axis at a unit distance from the
111  * origin, while the n-th vertex is the origin. With this convention,
112  * plugging in the coordinates of the standard vertices, we obtain:
113  * \f[
114  * \underbrace{\left[\begin{array}{lll}
115  * 1&0&0\\
116  * 0&1&0\\
117  * 1&1&1\\
118  * \end{array}
119  * \right]}_{L2X}
120  * \left[\begin{array}{l}
121  * \lambda_1\\
122  * \lambda_2\\
123  * \lambda_3\\
124  * \end{array}
125  * \right]=
126  * \left[\begin{array}{c}
127  * s_1\\
128  * s_2\\
129  * 1\\
130  * \end{array}
131  * \right]
132  * \f]
133  */
134  MatrixHigherDimD temp(MatrixHigherDimD::Identity());
135  temp.template block<1,dim>(dim,0)=Eigen::Matrix<double,1,dim>::Ones();
136  return temp;
137  }

149  /**********************************************************************/
150  static MatrixDimHigherDimD get_NdA()
151  {/*!/returns a matrix having in columns the non-unitary outward normals
152  * in the standard simplex. The norm of the normal is the ratio between
153  * the area of the face to the area of a dim-1 Simplex.
154  */
155  MatrixHigherDimDimD temp(-get_L2X().template block<dim+1,dim>(0,0));
156  for (int r=0;r<dim;++r)
157  {
158  temp.row(r).normalize(); // norm of first dim-1 normals is unitary
159  }
160  temp.row(dim)*=-1.0; // flip direction of the inclined face normal
161  return temp.transpose();
162  }
